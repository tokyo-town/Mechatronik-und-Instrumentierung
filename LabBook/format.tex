%======= Eine kompakte Umgebung für die Bilder
\newcommand{\bild}[5]{{
\begin{figure}[#1]
\begin{center}
\includegraphics[width=#2\linewidth]{#3}
\caption{#4}\label{#5}
\end{center}
\end{figure}
}}

\newcommand{\bilder}[7]{{
    \begin{figure}[#1]
    \begin{center}
    \begin{subfigure}{0.5\textwidth}
    \includegraphics[width=#2\linewidth]{#3}
    % \caption{#4}\label{#5}
    \end{subfigure}%
    \begin{subfigure}{0.5\textwidth}
    \includegraphics[width=#4\linewidth]{#5}
    \end{subfigure}
    \caption{#6}\label{#7}
    \end{center}
    \end{figure}
}}

\newcommand{\tabelle}[5]{{
\begin{table}[#1]
    \centering
    \caption{#2}
    \label{#3}
    \begin{tabular}{#4}
        #5
    \end{tabular}
\end{table}
}}
